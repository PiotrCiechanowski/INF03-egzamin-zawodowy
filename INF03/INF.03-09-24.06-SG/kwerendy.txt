SELECT liczbaDni, AVG(cena) AS sredniaCena FROM wycieczki GROUP BY liczbaDni;
SELECT wycieczki.miejsce, zdjecia.nazwa FROM wycieczki INNER JOIN zdjecia ON wycieczki.id = zdjecia.Wycieczki_id WHERE wycieczki.cena > 500;
